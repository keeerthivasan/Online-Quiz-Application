#include using namespace std; #define MAX 10 // total seats // Structure for booking details struct Ticket { int seatNo; string name; int isBooked; }; // Global array Ticket t[MAX]; // Initialize seats void initialize() { for (int i = 0; i < MAX; i++) { t[i].seatNo = i + 1; t[i].name = ""; t[i].isBooked = 0; } } // Display seats void displaySeats() { cout << "\nSeat Status:\n"; for (int i = 0; i < MAX; i++) { if (t[i].isBooked == 0) cout << "Seat " << t[i].seatNo << " -> Available\n"; else cout << "Seat " << t[i].seatNo << " -> Booked\n"; } } // Book ticket void bookTicket() { int seat; string name; cout << "Enter seat number to book: "; cin >> seat; if (seat < 1 || seat > MAX) { cout << "Invalid seat number!\n"; return; } if (t[seat - 1].isBooked == 1) { cout << "Seat already booked!\n"; } else { cout << "Enter your name: "; cin >> name; t[seat - 1].isBooked = 1; t[seat - 1].name = name; cout << "Ticket booked successfully!\n"; } } // Cancel ticket void cancelTicket() { int seat; cout << "Enter seat number to cancel: "; cin >> seat; if (seat < 1 || seat > MAX) { cout << "Invalid seat number!\n"; return; } if (t[seat - 1].isBooked == 0) { cout << "Seat is not booked!\n"; } else { t[seat - 1].isBooked = 0; t[seat - 1].name = ""; cout << "Ticket cancelled successfully!\n"; } } Quiz App
